- git init
- git remote add origin git@github.com:.../...
- git clone git@github.com:.../...
- git status
- git add .
- git commit -am " "  # a for add # skips git add (only for modified files)
- git commit -m " "
- git remote -v
- git push -u origin main
- git push
- git log (log of all commits)

# reset (undo)
- git reset
- git reset file
- git reset HEAD # undo to last (1 step back)
- git reset HEAD~1 # undo to last + 1 (2 steps back)
- git reset hash (hash found in git log)
- git reset --hard hash (hard change of all since hash)


# branch
- git branch
- git checkout -b exampleName
- git checkout nameofBranchtoSwitchTo
- git diff nameOfBranch
- git push -u origin exampleName
- git branch -d exampleName # to delete
- git merge exampleName # when on main

# pull
- git pull origin main
- git pull # if already origin main
